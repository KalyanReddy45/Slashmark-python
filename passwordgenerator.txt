import random
import string

def generate_password(length):
    """Generate a single password of a given length."""
    characters = string.ascii_letters + string.digits + string.punctuation
    return ''.join(random.choice(characters) for i in range(length))

def password_generator():
    num_passwords = int(input("How many passwords do you want to generate? "))
    print(f"Generating {num_passwords} password(s)")
    
    min_length = 3  # Minimum length of password
    for i in range(1, num_passwords + 1):
        length = int(input(f"Enter the length of Password #{i} (minimum {min_length}): "))
        if length < min_length:
            print(f"Password length should be at least {min_length}. Setting length to {min_length}.")
            length = min_length
        password = generate_password(length)
        print(f"Password #{i}: {password}")

# Start the password generator
password_generator()
